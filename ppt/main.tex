\begin{frame}{Scientific Question}
  \begin{block}{Main question}
    With only one to four optical spectra per newly observed supernova, what can we claim robustly?
  \end{block}
  \begin{itemize}
    \item Confirm type/subtype and spectral phase using template/classification context.
    \item Measure conservative line velocities, pseudo-equivalent widths, and line widths.
    \item Identify host contamination or narrow host-galaxy lines that affect interpretation.
    \item Compare each object with public sparse-spectra samples rather than over-modeling.
  \end{itemize}
\end{frame}

\begin{frame}{Literature Strategy}
  \begin{columns}[T,totalwidth=\linewidth]
    \begin{column}{0.48\linewidth}
      \begin{block}{Methods that fit sparse spectra}
        \begin{itemize}
          \item SNID / Superfit / DASH for type, redshift, and phase.
          \item BSNIP and CSP for Type Ia Si II / Ca II comparisons.
          \item Gutiérrez and Tsinghua Type II samples for Balmer/Fe II trends.
          \item Modjaz, Liu, and Xiang for stripped-envelope He/O/Ca checks.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\linewidth}
      \begin{block}{Interpretation boundary}
        \begin{itemize}
          \item Sparse spectra support classification and line diagnostics.
          \item They do not strongly constrain ejecta mass, explosion energy, or abundance tomography.
          \item TARDIS is used only as a qualitative line-ID and shape check.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{What Is Robust Now?}
  \begin{block}{Reasonable first-pass claims}
    \begin{itemize}
      \item Target-level type context and available phase coverage.
      \item Conservative adopted line velocities for primary features.
      \item pEW/FWHM trends as inspection-guided diagnostics.
      \item Host-line contamination warnings.
    \end{itemize}
  \end{block}
  \begin{block}{Still tentative}
    \begin{itemize}
      \item Velocity gradients from too few epochs.
      \item Subtype claims for stripped-envelope objects without He/O/Ca visual checks.
      \item Ejecta mass, explosion energy, or abundance structure from TARDIS alone.
    \end{itemize}
  \end{block}
\end{frame}
